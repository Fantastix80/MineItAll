CREATE DATABASE c_cpp_cinema;

USE c_cpp_cinema;

CREATE TABLE salle(
	numero int PRIMARY KEY,
    capacite int
);

CREATE TABLE film(
  	nom VARCHAR(50) PRIMARY KEY,
	duree int,
    salle int REFERENCES salles(numero)
);
